\begin{equation}
\left\{
\begin{array}{l}
\alpha_i = \widehat{z_{i-1},z_i} /x_{i-1} \\
d_i = \overline{O_{i-1},z_i} / x_{i-1} \\
r_i = \overline{x_{i-1},O_i} / z_i \\
\theta_i = \widehat{x_{i-1},x_i} /z_i \\
\end{array}
\right.
\label{DH}
\end{equation}
